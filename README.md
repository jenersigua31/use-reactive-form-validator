```bash
npm install --save use-reactive-form-validator
```

## Usage

```tsx
import * as React from 'react'

import { useMyHook } from 'use-reactive-form-validator'

const Example = () => {
  const example = useMyHook()
  return (
    <div>
      {example}
    </div>
  )
}
```
